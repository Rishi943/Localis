**finance.py — Add GET /finance/periods endpoint:**
```python
@router.get("/periods")
async def get_periods(request: Request):
    conn = database._connect_db()
    c = conn.cursor()
    c.execute("SELECT DISTINCT strftime('%Y-%m', date) as period FROM fin_transactions ORDER BY period ASC")
    periods = [r[0] for r in c.fetchall()]
    conn.close()
    return {"periods": periods}
```
